js:
document.querySelectorAll('[data-lightbox]').forEach(link => {
  link.addEventListener('click', e => {
    e.preventDefault();
    const src = link.getAttribute('href');
    const overlay = document.createElement('div');
    overlay.className = 'lightbox-overlay';
    overlay.innerHTML = `<img src="${src}" alt="Certificate preview">`;
    overlay.addEventListener('click', () => overlay.remove());
    document.body.appendChild(overlay);
  });
});

Result:
Clicking any certificate image opens it in a full-screen overlay.

Clicking anywhere on the overlay closes it.

Works with any image link that has data-lightbox.
